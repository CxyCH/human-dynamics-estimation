yarp_add_idl(STRING_MSGS_FILES String.msg)
yarp_add_idl(GEOMETRY_MSGS_FILES
    geometry_msgs/Quaternion.msg
    geometry_msgs/Transform.msg
    geometry_msgs/TransformStamped.msg
    geometry_msgs/Vector3.msg
    geometry_msgs/Wrench.msg
    geometry_msgs/WrenchStamped.msg)
yarp_add_idl(SENSOR_MSGS_FILES
    sensor_msgs/JointState.msg
    sensor_msgs/Temperature.msg)
yarp_add_idl(STD_MSGS_FILES std_msgs/Header.msg)
yarp_add_idl(TF_MSGS_FILES tf/tfMessage.msg)
yarp_add_idl(TF2_MSGS_FILES tf2_msgs/TFMessage.msg)

add_library(HumanROSMsgs
    ${STRING_MSGS_FILES}
    ${GEOMETRY_MSGS_FILES}
    ${SENSOR_MSGS_FILES}
    ${STD_MSGS_FILES}
    ${TF_MSGS_FILES}
    ${TF2_MSGS_FILES}
    String.msg
    geometry_msgs/Quaternion.msg
    geometry_msgs/Transform.msg
    geometry_msgs/TransformStamped.msg
    geometry_msgs/Vector3.msg
    geometry_msgs/Wrench.msg
    geometry_msgs/WrenchStamped.msg
    sensor_msgs/JointState.msg
    sensor_msgs/Temperature.msg
    std_msgs/Header.msg
    tf/tfMessage.msg
    tf2_msgs/TFMessage.msg)

set_target_properties(HumanROSMsgs PROPERTIES LINKER_LANGUAGE CXX)
target_include_directories(HumanROSMsgs PUBLIC
    $<BUILD_INTERFACE:${${PROJECT_BINARY_DIR}}/autogenerated/msgs/include>)

install(TARGETS HumanROSMsgs
        RUNTIME DESTINATION ${CMAKE_INSTALL_BINDIR}
        ARCHIVE DESTINATION ${CMAKE_INSTALL_LIBDIR}
        LIBRARY DESTINATION ${CMAKE_INSTALL_LIBDIR})
